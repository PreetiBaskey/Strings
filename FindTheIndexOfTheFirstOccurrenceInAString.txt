class Solution {
public:
    int strStr(string haystack, string needle) {
        int m = haystack.length();
        int n = needle.length();
        int j;
        for(int i = 0; i < m; i++) {
            for(j = 0; j < n; j++) {
                if(haystack[i + j] != needle[j]) {
                    break;
                }
            }
            if(j == n) {
                return i;
            }
        }

        return -1;
    }
};
