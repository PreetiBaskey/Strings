class Solution {
public:
    string makeGood(string s) {
        string result = "";
        int n = s.length();

        if(n == 1) {
            return s;
        }

        for(char &ch: s) {
            if(!result.empty() && abs(result.back() - ch) == 32) {
                result.pop_back();
            }
            else {
                result.push_back(ch);
            }
        }

        return result;
    }
};
