class Solution {
public:
    string removeDuplicateLetters(string s) {
        int n = s.length();
        string result;
        vector<bool> taken(26, false); //checks if char is taken or not
        vector<int> lastIndex(26); //stores the index of last occurrence of char

        for(int i = 0; i < n; i++) { //finding last index of char and store it
            char ch = s[i];
            lastIndex[ch - 'a'] = i;
        }

        //if last char of result is > than the current char then and it is coming again in the   future then pop it from the result
        for(int i = 0; i < n; i++) { 
            char ch = s[i];

            int idx = ch - 'a';
            if(taken[idx] == true) {
                continue;
            }

            while(result.length() > 0 && result.back() > ch && lastIndex[result.back() - 'a'] > i) {
                taken[result.back() - 'a'] = false;
                result.pop_back();
            }
            result.push_back(ch);
            taken[idx] = true;

        }

        return result;
    }
};
