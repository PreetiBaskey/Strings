class Solution {
public:
    int numberOfWays(string corridor) {
        int n = corridor.length();
        long long result = 1;
        vector<int> seatsIndices; //store index of all seats
        int MOD = 1e9 + 7;

        for(int i = 0; i < n; i++) {
            if(corridor[i] == 'S') {
                seatsIndices.push_back(i);
            }
        }

        //if seats count is either 0 or seats count is odd, then return 0
        if(seatsIndices.size() % 2 != 0 || seatsIndices.size() == 0) {
            return 0;
        }

        int prevEndSeatIdx = seatsIndices[1]; //index of first pair's second/last index
        for(int i = 2; i < seatsIndices.size(); i += 2) {
            int length = seatsIndices[i] - prevEndSeatIdx; //diff - prevlast - currfirst
            result = (result * length) % MOD;

            prevEndSeatIdx = seatsIndices[i + 1]; //make prevlast = curr[i + 1]
        }

        return (int)result;
    }
};
