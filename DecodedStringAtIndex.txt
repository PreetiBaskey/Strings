class Solution {
public:
    string decodeAtIndex(string s, int k) {
        int n = s.length();

        long long size = 0;
        //increasing/updating the size based on the the digit and alphabet, if digit(in char form), then multiply with it after changing it into a digit, if normal char, then simply add it by 1, as 1 char counts as +1
        for(char &ch: s) {
            if(isdigit(ch)) {
                size = size * (ch - '0');
            }
            else {
                size += 1;
            }
        }

        //1. better to traverse from back, because, it helps to get the original char, because it multiplied multiple times, firstly, we have to remove those
        for(int i = n - 1; i >= 0; i--) {
            k = k % size;

            if(k == 0 && isalpha(s[i])) { //if k = 0 and it is apha, then return char
                return string(1, s[i]);
            }
            if(isalpha(s[i])) { //if it is only aplha and not zero, then decrease size by 1
                size -= 1;
            }
            else { //else it is a digit, then divide it with the digit, but firstly convert it into a number, because we will get it in char form from the string - by ch - '0'
               size = size / (s[i] - '0');
            }
        }

        return "";  
    }
};
